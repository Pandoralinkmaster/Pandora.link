_startTimer() {

const s = (elapsed % 60).toString().padStart(2, '0');
document.getElementById('runtime').textContent = h + ':' + m + ':' + s;
}, 1000);
}

_processLoop() {

this.restarts++;

this._processLine(this.currentLine);
this.currentLine++;

this.processInterval = setTimeout(() => {
this._processLoop();
}, this.speed);
}

_processLine(idx) {
const line = this.sourceCode[idx];
const lineEl = document.getElementById('line-' + idx);
const statusEl = document.getElementById('status-' + idx);

if (!lineEl) return;

this._activateAlgoStep('read');
lineEl.classList.add('processing');
statusEl.textContent = 'verarbeite...';
statusEl.className = 'status processing';

lineEl.scrollIntoView({ behavior: 'smooth', block: 'center' });

setTimeout(() => {
this._activateAlgoStep('try');

const isBlocked = Math.random() < 0.15 && line.trim().length > 0;

if (isBlocked) {
this._handleBlockade(idx, line, lineEl, statusEl);
} else {
this._handleSuccess(idx, line, lineEl, statusEl);
}

document.getElementById('processedLines').textContent = idx + 1;
document.getElementById('codeCounter').textContent = (idx + 1) + '/' + this.sourceCode.length;

}, this.speed * 0.3);
}
